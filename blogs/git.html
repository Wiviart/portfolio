# Begin using git
git init

# Pull latest changes from origin
git pull

# Publish changes to origin
git push

# Commit all your local changes
git commit -a

# Fetch latest changes from origin
git fetch

# Files changed in working directory
git status

# A commit identified by $ID
git show $ID[:$file]

# All local branches (star "*" marks the current branch)
git branch

# Change between files
git diff [$id1] [$id2]

# Who is the changer
git blame $file

# History of changes
git log [-p $file $directory]

# Push to branch, auto-create if not exist
git push origin HEAD:<branch_name>

# Undo git pull
git reflog
git reset --hard <index>
# or
git merge --abort
# or
git reset --merge

# Rename and track remote branch
git branch -m test main
git fetch origin
git branch -u origin/main main
git remote set-head origin -a
